ACTOR ItemStunner : CustomInventory 15000
{
	Radius 16
	Height 8
	Scale 0.5
	Inventory.PickupMessage "You got the stun gun." 
	States
	{
	Spawn:
		TAZI A 1
		Loop
	Pickup: 
		TAZI A 0 
		TAZI A 0 A_PlaySound("pickups/weapon")
		TAZI A 0 A_GiveInventory("Stunner",1)
		Stop
	}
}

ACTOR Stunner : Weapon
{
	Weapon.SelectionOrder 450
	Weapon.AmmoType "Cell"
	Weapon.AmmoGive 50
	Weapon.AmmoUse 2	
	+Weapon.NoAlert
	+NoAutoFire
	Weapon.SlotNumber 1
	States
	{
		Ready:
			TAZR A 1 A_WeaponReady
			Loop
		Select:
			TAZR A 1 A_Raise
			Loop
		Deselect:
			TAZR A 1 A_Lower
			Loop
		Fire: //Stun
			TAZR B 1 A_WeaponReady
			TAZR B 18 A_Saw("","weapons/sawhit", 3, "StunnerDummy")
			TAZR B 0 A_Refire
			Goto Ready
		AltFire: //Melee
			TAZR B 18 A_CustomPunch(3)
			TAZR A 8
			Goto Ready
		
	}
}
