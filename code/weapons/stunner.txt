ACTOR ItemStunner : CustomInventory 15000
{
	Radius 16
	Height 8
	Scale 0.5
	Inventory.PickupMessage "You got the stun gun." 
	States
	{
	Spawn:
		TAZI A 1
		Loop
	Pickup: 
		TAZI A 0 
		TAZI A 0 A_PlaySound("pickups/weapon")
		TAZI A 0 A_GiveInventory("Stunner",1)
		Stop
	}
}

ACTOR Stunner : Weapon
{
	Weapon.SelectionOrder 450
	Weapon.AmmoType "Cell"
	Weapon.AmmoGive 50
	Weapon.AmmoUse 2	
	+Weapon.NoAlert
	+NoAutoFire
	Weapon.SlotNumber 1
	States
	{
		Ready:
			TAZR A 2
			TAZR B 3
			TAZR C 3
			TAZR D 4
			Goto Bobbing
		Bobbing: //The 'real' ready state
			TAZR D 2 A_WeaponReady
			Loop
		Select:
			TAZR A 0 A_Raise
			TAZR A 1 A_Raise
			Loop
		Deselect:
			TAZR D 5
			TAZR C 3
			TAZR B 3
			TAZR A 2
			Goto Returned
		Returned:
			TNT1 A 1 A_Lower
			Loop
		Fire: //Stun
			TAZR D 3 
			TAZR E 3 A_WeaponReady
			TAZR F 18 A_Saw("","weapons/sawhit", 3, "StunnerDummy")
			TAZR E 3
			TAZR E 0 A_Refire
			Goto Bobbing
		AltFire: //Melee
			TAZR D 3
			TAZR G 4
			TAZR H 4
			TAZR I 2
			TAZR J 3 A_CustomPunch(3)
			TAZR D 3
			Goto Bobbing
		
	}
}
