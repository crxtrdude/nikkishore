ACTOR NikkiFist: Fist Replaces Fist
{
	+CHEATNOTWEAPON
	+NOAUTOFIRE
	States
	{
		Spawn:
			TNT1 A -1
			Loop
		Ready:
			NFST A 5 A_WeaponReady
			Goto Ready
		Deselect:
			NFST A 1 A_Lower
			Loop
		Select:
			NFST A 1 A_Raise
			Loop
		Fire:
			//FIST
			NFST F 3 A_PlaySound("world/swing")
			NFST G 2
			NFST H 6 A_CustomPunch(2, 0, 0, "MeleePuff")
			NFST F 4
			NFST A 3
			NFST A 0 A_ReFire
			Goto Ready
		AltFire:
			NFST A 0 A_JumpIfInventory("DoubleTapCheck",1,13)
			NFST A 0 A_GiveInventory("DoubleTapCheck",1)
			NFST I 8 A_WeaponReady
			NFST A 0 A_TakeInventory("DoubleTapCheck",999)
			NFST I 0 
			NFST J 2 
			NFST J 1 
			NFST K 7 A_CustomPunch(4, 0, 0, "BulletPuff")
			NFST I 2
			NFST I 0 A_ReFire
			Goto Ready
			NFST F 2 A_FireCustomMissile ("SledgeHammerDummy", 0, 0)
			NFST L 2 SetPlayerProperty(0,1,0)
			NFST M 2 
			NFST M 0 SetPlayerProperty(0,0,0)
			NFST M 0 SetPlayerProperty(0,1,4)
			NFST N 1 A_PlaySound("world/swing")
			NFST O 1 
			NFST P 1 A_CustomPunch(random(35,40), 1, 0, "BulletPuff", 58)
			NFST P 10
			NFST O 5 SetPlayerProperty(0,0,4)
			NFST A 0 A_TakeInventory("DoubleTapCheck",999)
			Goto Ready
	}
}

ACTOR DoubleTapCheck : Ammo
{
	inventory.maxamount 1
	ammo.backpackamount 0
	ammo.backpackmaxamount 0
	inventory.icon TNT1A0
}
