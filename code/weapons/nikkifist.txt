ACTOR NikkiFist: Fist Replaces Fist
{
	+CHEATNOTWEAPON
	+NOAUTOFIRE

	States
	{
		Spawn:
			TNT1 A -1
			Loop
		Ready:
			//AFST A 0 A_Jump(192, 9) - JUMPS FROM KNUCKLE POP 
			AFST A 5 A_WeaponReady
			Goto Ready
		Deselect:
			AFST A 1 A_Lower
			Loop
		Select:
			AFST A 1 A_Raise
			Loop
		Fire:
			//FIST
			AFST F 3 A_PlaySound("world/swing")
			AFST G 2
			AFST H 4 A_CustomPunch(2, 0, 0, "MeleePuff")
			AFST G 4
			AFST A 3
			AFST A 0 A_ReFire
			Goto Ready + 9
		AltFire:
			AFST A 0 A_JumpIfInventory("DoubleTapCheck",1,13)
			AFST A 0 A_GiveInventory("DoubleTapCheck",1)
			AFST I 8 A_WeaponReady
			AFST A 0 A_TakeInventory("DoubleTapCheck",999)
			AFST I 0 
			AFST J 2 A_PlaySound("world/swing")
			AFST K 1 
			AFST L 1
			AFST M 7 A_CustomPunch(4, 0, 0, "MeleePuff")
			AFST L 4
			AFST K 3
			AFST J 1 
			AFST J 0 A_ReFire
			Goto Ready + 9
			AFST A 2 A_FireCustomMissile ("SledgeHammerDummy", 0, 0)
			UCUT A 2 SetPlayerProperty(0,1,0)
			UCUT B 2 
			UCUT C 2 
			UCUT D 1 
			UCUT E 1 
			UCUT F 0 SetPlayerProperty(0,0,0)
			UCUT F 0 SetPlayerProperty(0,1,4)
			UCUT F 1 A_PlaySound("world/swing")
			UCUT G 1 
			UCUT H 1 A_CustomPunch(random(35,40), 1, 0, "UppercutPuff", 58)
			UCUT I 2 
			UCUT J 12 
			UCUT I 5 SetPlayerProperty(0,0,4)
			UCUT H 3 
			UCUT G 2 
			UCUT K 2 
			UCUT A 0 A_TakeInventory("DoubleTapCheck",999)
			Goto Ready + 9
	}
}

ACTOR DoubleTapCheck : Ammo
{
inventory.maxamount 1
ammo.backpackamount 0
ammo.backpackmaxamount 0
inventory.icon TNT1A0
}
